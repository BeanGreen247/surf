  char result[BUFSIZE];

  assert (fd >= 0);
  assert (fstat (fd, &stat_buf) == 0);

  if (stat_buf.st_size > 0 && S_ISREG (stat_buf.st_mode))

      FILE *f = fdopen (fd, "r");

}
